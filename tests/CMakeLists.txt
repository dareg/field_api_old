	add_executable(${TEST_NAME}.x ${TEST_FILE})
	target_link_libraries(${TEST_NAME}.x ${LIBNAME} ${LNK_FLAGS})
	add_test(NAME ${TEST_NAME} COMMAND ${TEST_NAME}.x)

endforeach()
